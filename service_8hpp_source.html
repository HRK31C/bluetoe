254  details::attribute_access_result details::attribute_access_impl< UUID >::attribute_access( details::attribute_access_arguments& args, std::uint16_t attribute_handle )
255  {
256  if ( args.type == details::attribute_access_type::read )
257  {
258  if ( args.buffer_offset > sizeof( uuid::bytes ) )
259  return details::attribute_access_result::invalid_offset;
260 
261  args.buffer_size = std::min< std::size_t >( sizeof( uuid::bytes ) - args.buffer_offset, args.buffer_size );
262 
263  std::copy( std::begin( uuid::bytes ) + args.buffer_offset , std::begin( uuid::bytes ) + args.buffer_offset + args.buffer_size, args.buffer );
264 
265  return args.buffer_size == sizeof( uuid::bytes ) - args.buffer_offset
266  ? details::attribute_access_result::success
267  : details::attribute_access_result::read_truncated;
268  }
269  else if ( args.type == details::attribute_access_type::compare_value )
270  {
271  if ( sizeof( uuid::bytes ) == args.buffer_size
272  && std::equal( std::begin( uuid::bytes ), std::end( uuid::bytes ), &args.buffer[ 0 ] ) )
273  {
274  return details::attribute_access_result::value_equal;
275  }
276  }
277 
278  return details::attribute_access_result::write_not_permitted;
