452  // transmit_ == transmit_end_ without the buffer beeing empty
453  if ( transmit_end_ >= transmit_ )

500  write_buffer ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::next_transmit()
501  {
502  // if transmit buffer is empty, create a new, empty one
503  if ( transmit_ == transmit_end_ )
504  {
505  transmit_ = transmit_buffer();
506  transmit_end_ = transmit_ + 2;
507  transmit_[ 0 ] = ll_empty_id;
508  transmit_[ 1 ] = 0;
509  }
510  else
511  {

513 
514  if ( more_data )

516  }
517 
518  // insert the next expected sequence for every attempt to send the PDU, because it could be that
519  // the slave is able to receive data, while the master is not able to.
520  transmit_[ 0 ] = ( transmit_[ 0 ] & ~nesn_flag ) | ( next_expected_sequence_number_ ? nesn_flag : 0 );

581  template < std::size_t TransmitSize, std::size_t ReceiveSize, typename Radio >
582  bool ll_data_pdu_buffer< TransmitSize, ReceiveSize, Radio >::check_receive_buffer_consistency( const char* label ) const
583  {
584  if ( received_ == received_end_ )
585  return true;
586 
587  const std::uint8_t* c = received_;
588 
589  if ( received_ < received_end_ )
590  {
591  // now, by following the length field, we have to reach received_end_ whithout wrapping
592  while ( c < received_end_ )
593  c += c[ 1 ] + ll_header_size;
594 
595  return c == received_end_;
596  }
597 
598  // now by following the length field, we have to reach the end of the buffer
599  while ( c != receive_buffer() && c < end_receive_buffer() )
600  {
601  c += c[ 1 ] + ll_header_size;
602 
603  if ( c > end_receive_buffer() )
604  return false;
605 
606  if ( end_receive_buffer() - c < 2 || c[ 1 ] == 0 )
607  c = receive_buffer();
608  }
609 
610  if ( c != receive_buffer() )
611  return false;
612 
613  // wrap around and have to meet received_end_
614  while ( c < received_end_ )
615  c += c[ 1 ] + ll_header_size;
616 
617  return c == received_end_;
618  }
